---
title: "StockX Product Activity"
api: "GET https://app.retailed.io/api/v1/stockx/product/activity"
description: "Get StockX product activity"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Retailed private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="nike-air-max-1-travis-scott-wheat">
    Product query : `air-jordan-3-retro-white-cement-reimagined` or `c318bbcc-312a-4396-9252-698c203d1dea`
</ParamField>

<ParamField query="type" type="string" optional={false} placeholder="bid">
    Type : `ask`, `bid`, or `sale`
</ParamField>

<ParamField query="country" type="string" optional={false} placeholder="US">
    Country code : `US` or `FR` or `GB`
</ParamField>

<ParamField query="currency" type="string" optional={false} placeholder="USD">
    Currency code : `USD` or `EUR` or `GBP`
</ParamField>

### Response

<ResponseField name="object" type="Product Object">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="bids" type="list">
            <Expandable title="properties" defaultOpen={true}>
                <ResponseField name="amount" type="string">
                    Amount of the bid.
                </ResponseField>
                <ResponseField name="count" type="string">
                    Number of bids.
                </ResponseField>
                <ResponseField name="size" type="string">
                    Size of the bid.
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="asks" type="list">
            <Expandable title="properties" defaultOpen={true}>
                <ResponseField name="amount" type="string">
                    Amount of the ask.
                </ResponseField>
                <ResponseField name="count" type="string">
                    Number of asks.
                </ResponseField>
                <ResponseField name="size" type="string">
                    Size of the ask.
                </ResponseField>
            </Expandable>
        </ResponseField>
        <ResponseField name="sales" type="list">
            <Expandable title="properties" defaultOpen={true}>
                <ResponseField name="amount" type="string">
                    Amount of the sale.
                </ResponseField>
                <ResponseField name="size" type="string">
                    Size of the sale.
                </ResponseField>
                <ResponseField name="created_at" type="string">
                    Date of the sale.
                </ResponseField>
            </Expandable>
        </ResponseField>
    </Expandable>
</ResponseField>
