---
title: "StockX Trends"
api: "GET https://app.retailed.io/api/v1/stockx/trends"
description: "Get the latest trends from StockX"
---

### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Retailed private key to use.
</ParamField>

### Parameters

<ParamField query="query" type="string" optional={false} placeholder="sneakers">
    Category query : `sneakers`, `streetwear`, `electronics`, `trading cards`, `collectibles`, `apparel`, `handbags`, `watches`
</ParamField>

<ParamField query="sort_by" type="string" optional={true} placeholder="release_date">
    Sort by : `featured`, `most_active` or `release_date`
</ParamField>

<ParamField query="country" type="string" optional={false} placeholder="US">
    Country code : `US` or `FR` or `GB`
</ParamField>


### Response

<ResponseField name="list" type="Products List">
    <Expandable title="properties" defaultOpen={true}>
        <ResponseField name="id" type="string">
            StockX product id
        </ResponseField>
        <ResponseField name="slug" type="string">
            StockX product slug
        </ResponseField>
        <ResponseField name="name" type="string">
            Product name
        </ResponseField>
        <ResponseField name="sku" type="string">
            Product SKU
        </ResponseField>
        <ResponseField name="brand" type="string">
            Product brand
        </ResponseField>
        <ResponseField name="image" type="string">
            Product image
        </ResponseField>
        <ResponseField name="colorway" type="string">
            Product colorway
        </ResponseField>
    </Expandable>
</ResponseField>

```bash trends
curl --location --request GET 'https://app.retailed.io/api/v1/stockx/trends?query=sneakers&sort_by=most_active' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
