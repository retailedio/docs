### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Retailed private key to use.
</ParamField>

### Query Parameters

<ParamField query="id" type="string" optional={true} placeholder="b5242fa5-48a4-444f-bd08-6a5db282b97b">
    Retailed internal ID
</ParamField>

<ParamField query="sku" type="string" optional={true} placeholder="DV6773-220">
    Sneaker code SKU
</ParamField>

<ParamField query="product_id" type="string" optional={true} placeholder="2f0ad3d7-ea60-423c-923a-d82372f97da8">
    Retailed internal product ID
</ParamField>

<ParamField query="size_eu" type="string" optional={true} placeholder="39.5">
    Sneaker size in EU
</ParamField>

<ParamField query="size_us" type="string" optional={true} placeholder="6">
    Sneaker size in US
</ParamField>

<ParamField query="size_jp" type="string" optional={true} placeholder="24.5">
    Sneaker size in JP
</ParamField>

<ParamField query="size_uk" type="string" optional={true} placeholder="6">
    Sneaker size in UK
</ParamField>

<ParamField query="csv" type="boolean" optional={true} placeholder="false">
    Export results to csv
</ParamField>
