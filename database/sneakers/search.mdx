---
title: "Search"
api: "GET https://app.retailed.io/api/v1/sneakers"
description: "Search for sneakers"
---

<Note>
    We provide our database in .csv export or available on a hyper-relevant search engine with Meilisearch.
    Contact us if you wish to synchronise with us.
</Note>


### Header

<ParamField header="x-api-key" type="string" required={true}>
    Parameter defines the Retailed private key to use.
</ParamField>

### Query Parameters

<ParamField query="id" type="string" optional={true} placeholder="b5242fa5-48a4-444f-bd08-6a5db282b97b">
    Retailed internal ID
</ParamField>

<ParamField query="name" type="string" optional={true} placeholder="Jordan">
    Sneaker name
</ParamField>

<ParamField query="sku" type="string" optional={true} placeholder="DV6773-220">
    Sneaker code SKU
</ParamField>

<ParamField query="colorway" type="string" optional={true} placeholder="Violet">
    Sneaker colorway
</ParamField>

<ParamField query="initial_price" type="integer" optional={true} placeholder="225">
    Sneaker initial price
</ParamField>

<ParamField query="sizing" type="string" optional={true} placeholder="woman">
    Sneaker sizing: `newbalance(w)`, `newbalance`, `woman`, `gradeschool`, `primaryschool`, `man`, `yeezy`, `yeezykids`, `crocs`, `converse`, `adidas`, `puma`.
</ParamField>

<ParamField query="brand_id" type="string" optional={true} placeholder="2bd9a45a-683b-11ed-9022-0242ac120002">
    Retailed internal brand ID
</ParamField>

<ParamField query="released_at" type="string" optional={true} placeholder="2021-07-27">
    Sneaker released at.
</ParamField>

<ParamField query="extended" type="boolean" optional={true} placeholder="true">
    Show more informations about the sneakers.
</ParamField>

<ParamField query="csv" type="boolean" optional={true} placeholder="false">
    Export results to csv.
</ParamField>

### Response

<ResponseField name="count" type="number">
    Indicates the number of sneakers returned
</ResponseField>

<ResponseField name="data" type="object">
    Array of sneakers
</ResponseField>

<ResponseField name="error" type="object">
    Not empty if an error occured
</ResponseField>

<ResponseField name="usage" type="object">
    API usage
</ResponseField>

<RequestExample>

```bash Example Request
curl --location --request GET 'https://app.retailed.io/api/v1/sneakers?name=Jordan 4 Retro Where the Wild Things Are&extended=true' \
--header 'Content-Type: application/json' \
--header 'x-api-key: <token>' \
```

</RequestExample>
