<ParamField query="query" type="string" optional={false} placeholder="nike-air-max-1-travis-scott-wheat">

</ParamField>

<ParamField query="start_date" type="string" optional={false} placeholder="2022-01-01">

</ParamField>

<ParamField query="end_date" type="string" optional={false} placeholder="2022-05-29">
    End date : `2021-05-29`
</ParamField>

<ParamField query="country" type="string" optional={false} placeholder="US">

</ParamField>

<ParamField query="currency" type="string" optional={false} placeholder="USD">

</ParamField>
